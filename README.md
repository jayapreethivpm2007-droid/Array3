# Array (list) to store marks
marks = [78, 85, 90, 67, 88]

# Display marks
print("Student Marks:", marks)

# Calculate total
total = sum(marks)

# Calculate average
average = total / len(marks)

print("Total Marks:", total)
print("Average Marks:", average)

Output:
Student Marks: [78, 85, 90, 67, 88]
Total Marks: 408
Average Marks: 81.6
